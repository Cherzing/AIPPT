# AIPPT Universal Native Editing Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a strict, staged migration path that lets AIPPT templates, imported PPT files, and user-made templates gain native PPT-style editing without breaking current generation, template creation, PPT import, PPTX export, or PDF export.

**Architecture:** Do not directly replace the existing React/CSS template system. Add a versioned native slide model (`content.__aippt`) and a capability layer that classifies each slide as high-fidelity native, auto-convertible native, imported-background native, or legacy-only. Use `PPTist/` as the main reference for editor interaction/model behavior, and `presenton/` as the reference for existing generation/template/import/export compatibility.

**Tech Stack:** Next.js 16, React 19, Redux Toolkit, TypeScript, PptxGenJS, LibreOffice PDF conversion, local reference repos `E:\Files\Work\2026-06-30 电厂三期 其他\PPT\PPTist` and `E:\Files\Work\2026-06-30 电厂三期 其他\PPT\presenton`.

---

## Decision Principles

- Do not promise every template will immediately match the coal-power template fidelity.
- Do not remove the current React/CSS renderer until native parity is measured and accepted.
- Do not enable native editor for a template unless the slide has a valid native model or an explicitly accepted fallback model.
- Do not export through native PPTX if any slide is still legacy-only.
- Do not use browser CSS as the source of truth for PPTX output.
- Do add explicit model versioning, conversion status, source renderer, and fidelity level.

---

## Fidelity Levels

Every slide must be classified into one of these levels:

- **A / High-Fidelity Native**: Editable native elements; PPTX/PDF export should closely match web preview. Coal-power template belongs here.
- **B / Structured Native**: Editable native elements generated by conversion; common text/image/shape/table layouts are editable, but exact visual parity is not guaranteed.
- **C / Imported Background Native**: Original slide is preserved as a locked background image; extracted text/images may be editable overlays.
- **D / Legacy Only**: React/CSS rendering only; limited text replacement; export uses legacy fallback.

The product UI must not imply Level B/C/D templates have the same fidelity as Level A.

```ts
import type { AipptSlideDocument } from './types';

export type NativeValidationResult =
  | { valid: true; document: AipptSlideDocument }
  | { valid: false; errors: string[] };

const FIDELITY = new Set(['A', 'B', 'C', 'D']);
const SOURCE = new Set(['native', 'coal-power-builder', 'legacy-template-converter', 'ppt-importer', 'manual-template-authoring']);
const STATUS = new Set(['complete', 'partial', 'background-fallback', 'legacy-only']);

export function validateNativeSlideDocument(value: unknown): NativeValidationResult {
  const errors: string[] = [];
  if (!value || typeof value !== 'object') return { valid: false, errors: ['document must be an object'] };
  const document = value as Partial<AipptSlideDocument>;

  if (document.width !== 1280) errors.push('width must be 1280');
  if (document.height !== 720) errors.push('height must be 720');
  if (!Array.isArray(document.elements)) errors.push('elements must be an array');
  if (!document.meta || typeof document.meta !== 'object') {
    errors.push('meta.version is required');
    errors.push('meta.fidelity is required');
    errors.push('meta.sourceRenderer is required');
    errors.push('meta.conversionStatus is required');
  }
  else {
    if (document.meta.version !== 1) errors.push('meta.version is required');
    if (!FIDELITY.has(document.meta.fidelity)) errors.push('meta.fidelity is invalid');
    if (!SOURCE.has(document.meta.sourceRenderer)) errors.push('meta.sourceRenderer is invalid');
    if (!STATUS.has(document.meta.conversionStatus)) errors.push('meta.conversionStatus is invalid');
  }

  if (errors.length > 0) return { valid: false, errors };
  return { valid: true, document: document as AipptSlideDocument };
}

export function isNativeSlideDocument(value: unknown): value is AipptSlideDocument {
  return validateNativeSlideDocument(value).valid;
}
```

```ts
export type TemplateInventoryLevel = 'A' | 'B' | 'C' | 'D';

export type TemplateInventoryResult = {
  level: TemplateInventoryLevel;
  reason: string;
  supportedFields: string[];
  unsupportedFields: string[];
};

const UNSUPPORTED_KEYS = new Set(['html', 'jsx', 'code', 'componentCode', 'reactCode']);
const SUPPORTED_PRIMITIVE_KEYS = new Set(['title', 'subtitle', 'subTitle', 'description', 'summary', 'caption', 'items', 'image', 'imageUrl', 'img', 'photo', 'logo']);

export function classifyTemplateContent(content: Record<string, unknown>): TemplateInventoryResult {
  const keys = Object.keys(content);
  const unsupportedFields = keys.filter((key) => UNSUPPORTED_KEYS.has(key));
  if (unsupportedFields.length > 0) {
    return { level: 'D', reason: 'contains raw code/html fields', supportedFields: [], unsupportedFields };
  }

  const supportedFields = keys.filter((key) => SUPPORTED_PRIMITIVE_KEYS.has(key) || typeof content[key] === 'string' || typeof content[key] === 'number' || Array.isArray(content[key]));
  const complexFields = keys.filter((key) => !supportedFields.includes(key));

  if (supportedFields.length > 0 && complexFields.length === 0) {
    return { level: 'B', reason: 'structured fields can be converted', supportedFields, unsupportedFields: [] };
  }

  if (supportedFields.length > 0) {
    return { level: 'B', reason: 'partially structured fields can be converted with warnings', supportedFields, unsupportedFields: complexFields };
  }

  return { level: 'D', reason: 'no supported structured fields', supportedFields: [], unsupportedFields: keys };
}
```

```ts
import { classifyTemplateContent, type TemplateInventoryLevel } from './template-inventory';
import { isNativeSlideDocument } from './native-schema';

export type SlideLike = {
  layout?: string | null;
  layout_group?: string | null;
  content?: Record<string, unknown> | null;
};

export type SlideNativeCapability = {
  level: TemplateInventoryLevel;
  mode: 'native' | 'convertible' | 'imported-background' | 'legacy-only';
  reason: string;
};

export function isCoalPowerLayout(slide: SlideLike): boolean {
  const layoutGroup = typeof slide.layout_group === 'string' ? slide.layout_group : '';
  const layout = typeof slide.layout === 'string' ? slide.layout : '';
  return layoutGroup === 'taicang-coal-power-report' || layout.includes('coal-power-');
}

export function getSlideNativeCapability(slide: SlideLike): SlideNativeCapability {
  const content = slide.content && typeof slide.content === 'object' ? slide.content : {};
  if (isNativeSlideDocument(content.__aippt)) return { level: content.__aippt.meta.fidelity, mode: 'native', reason: 'valid native model exists' };
  if (isCoalPowerLayout(slide)) return { level: 'A', mode: 'native', reason: 'dedicated coal-power native builder exists' };
  if (typeof content.__imported_background === 'string') return { level: 'C', mode: 'imported-background', reason: 'imported slide background fallback exists' };
  const inventory = classifyTemplateContent(content);
  if (inventory.level === 'B') return { level: 'B', mode: 'convertible', reason: inventory.reason };
  return { level: 'D', mode: 'legacy-only', reason: inventory.reason };
}

export function canUseNativeEditor(slide: SlideLike): boolean {
  return getSlideNativeCapability(slide).level !== 'D';
}
```
